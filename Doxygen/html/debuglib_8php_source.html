377  // remove leading tabs
378  public static function trim_leading_tabs( $string, $tab_padding = NULL ) {
379  /* remove whitespace lines at the start and end of the string */
380  $string = preg_replace(array('/^\s*\n/', '/\s*$/'), array('',''), $string);
381 
382  // find the smallest leading tab count
383  preg_match_all('/^\t+/', $string, $matches);
384 
385  if(count($matches[0]) > 0) {
386  $min_tab_count = strlen(min($matches[0]));
387 
388  // und entfernen
389  $string = preg_replace('/^\t{'.$min_tab_count.','.$min_tab_count.'}/m', (isset($tab_padding) ? str_repeat("\t", $tab_padding) : ''), $string);
390  }
391 
392  return $string;
393  }

395  public static function _handle_whitespace( $string ) {
396  // replace 2 or more spaces with nobreaks (for special markup)
397  $string = preg_replace_callback(
398  '/ {2,}/',
399  create_function(
400  '$matches',
401  'return str_repeat("&nbsp;", strlen($matches[0]));'
402  ),
403  $string
404  );
405 
406  $string = preg_replace(array('/&nbsp;$/', '/^&nbsp;/'), '<span class="DbugL_outer_space">&nbsp;</span>', $string); # mark spaces at the start/end of the string with red underscores
407  $string = str_replace("\t", '&nbsp;&nbsp;<span class="DbugL_tabs">&nbsp;</span>', $string); # replace tabulators with ' �'
408  return $string;
409  }

411  // format strings for output to html
412  public static function format_string($string, $trim_tabs = NULL, $escape_html = 2) {
413 
414  $escape_html == 2 and $string = htmlspecialchars($string);
415  if($escape_html == 0) return $string;
416 
417  $is_multiline = strpos($string, "\n") !== FALSE;
418 
419  if($is_multiline && isset($trim_tabs)) {
420  $string = self::trim_leading_tabs($string, $trim_tabs);
421  }
422 
423  $string = self::_handle_whitespace($string);
424  $string = nl2br($string);
425  if($is_multiline) {
426  $string = '<span class="DbugL_multi">'.$string.'</span>';
427  }
428 
429  return $string;
430  }

432  // parse the options string in css syntax
433  public static function parse_options($options_string = NULL, $alt_parameter_names = array()) {
434 
435  $options = array();
436 
437  $alt_parameter_mapping = array();
438  foreach($alt_parameter_names as $parameter_name => $alt_names) {
439  $alt_parameter_mapping[$parameter_name] = $parameter_name;
440  foreach($alt_names as $alt_name) {
441  $alt_parameter_mapping[$alt_name] = $parameter_name;
442  }
443  }
444 
445  if(!$options_string) return $options;
446 
447  if(strpos($options_string, ':') !== FALSE) {
448  $pairs = explode(';', $options_string);
449 
450  foreach($pairs as &$pair) {
451 
452  $pair = trim($pair);
453  if($pair == '') continue;
454 
455  if(strpos($pair, ':') !== FALSE) {
456 
457  list($option, $value) = explode(':', $pair);
458 
459  if(isset($alt_parameter_mapping[$option])) {
460  $options[$alt_parameter_mapping[$option]] = $value;
461  } else {
462  $options[$option] = $value;
463  }
464 
465  }
466 
467  }
468  } else { // if the options string has no ":" in it, consider it to be the label
469  $options['label'] = $options_string;
470  }
471 
472  return $options;
473  }

501  for($i=0; $i<6; $i+=2) {
502  $component = substr( $base_color, $i, 2 );
